Integration of the service into your website:
1. Add JavaScript in your HTML-page: 
<script src="https://steeimages.com/inject.js"></script>
2. Add button in your HTML-page: 
<button id="upload">Upload to IPFS</button>
3. Add eventListener for upload button: 
<script>
document.getElementById('upload').addEventListener('click', function() {
    uploadImageToIpfs(function( err, files ) {
        if (err) console.error(`IPFS error: ${err}`)
        else console.log(files[0][0])
    })
});
</script>
4. You can set remote addresses to your IPFS node: 
<script>
let ChangeNode = {}
function initCustomConnection() {
    ChangeNode.connectionCustom = {
        api : {
            protocol: `http`,
            port: `5001`,
            address: `13.59.234.79`
        },
        gateway: {
            protocol: `http`,
            port: `7777`,
            address: `13.59.234.79`
        }
    }
    initConnection(ChangeNode.connectionCustom);
}
</script>
